* font

#+BEGIN_SRC emacs-lisp :tangle yes
  (when (member "xos4 Terminus" (font-family-list))
    (set-face-attribute 'default nil :font "xos4 Terminus"))
  (set-face-attribute 'default nil :height 166)
#+END_SRC
